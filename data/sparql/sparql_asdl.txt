identifier, column_id, subject, object, predicate
query_anatomy = QUERY(select_expr select, where_expr where, order_expr? order)
select_expr = SELECT(select_col* select_val)
select_col = SELECT_COL(agg_op? agg_op_val, col_type column_type)
col_type = DISTINCT(column_id column) | CONST(column_id column)
agg_op = SUM | COUNT | MIN | MAX | AVG
desc = DESC
where_expr = WHERE(where_col* where_val)
where_col = WHERE_COL(subject subject_val, predicate predicate_val, object object_val)
order_expr = ORDERBY(desc? desc_flag, column_id ord_column)
